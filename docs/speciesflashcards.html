Species Flash Cards

Enter the species you want below, separated by commas. An example is Leopard frog, Hyla avivoca, Hyla chrysoscelis, Hyla gratiosa. You can use common names, scientific names, or a mixture. Or you can enter a single URL, such as https://www.nps.gov/grsm/learn/nature/fish-checklist.htm, and it will attempt to pull all species names (scientific names only) from that page. Then click the Store image locations button. It will take some time to get these locations — you should see a progress bar. After that, New image will give you an image to identify; mouseover the image to see the species and common name.

This is made possible due to iNaturalist.org and all their users who make observations and help identify species, but it is in no way affiliated with them; if this app is having issues, please contact Brian O'Meara, not the good iNaturalist folks. Scientific name extraction from web pages is provided by Global Names using their API to their gnfinder tool. The first draft of this was done to help my daughter study for a species identification quiz in middle school. It was converted from R by Claude Opus 4.6 in order to preserve the functionality given then impending shutdown of shinyapps.io using the prompt "Convert the original R code to a quarto document that just uses javascript only" on April 5, 2026. 
// =========================================================================
// Species Flash Cards — Pure JavaScript (no R, no webR)
//
// Replaces:
//   rinat::get_inat_obs()       → fetch() to iNaturalist API v1
//   jsonlite + R string ops     → fetch() to Global Names API + JS parsing
//   Shiny reactive UI           → direct DOM manipulation
// =========================================================================

app = {

    // --- State ---
    let imageList = [];

    // --- DOM refs ---
    const queryButton       = document.getElementById('queryButton');
    const goButton          = document.getElementById('goButton');
    const speciesArea       = document.getElementById('speciesArea');
    const pictureEl         = document.getElementById('picture');
    const creditEl          = document.getElementById('credit');
    const loadedEl          = document.getElementById('loaded');
    const introEl           = document.getElementById('intro');
    const progressContainer = document.getElementById('progress-container');
    const progressFill      = document.getElementById('progress-fill');
    const progressText      = document.getElementById('progress-text');

    // =====================================================================
    // iNaturalist API v1 — replaces rinat::get_inat_obs()
    //
    // Original R:
    //   obs <- rinat::get_inat_obs(query=species[i], maxresults=50)
    //   obs <- obs[nchar(obs$image_url)>5,]
    //   obs <- subset(obs, quality_grade=="research")
    // =====================================================================
    async function getINatObs(query, maxResults = 50) {
        const url =
            'https://api.inaturalist.org/v1/observations?' +
            'q=' + encodeURIComponent(query) +
            '&quality_grade=research' +
            '&photos=true' +
            '&per_page=' + maxResults +
            '&order=desc&order_by=id';
        try {
            const resp = await fetch(url);
            if (!resp.ok) return [];
            const data = await resp.json();
            if (!data.results || data.results.length === 0) return [];

            return data.results.filter(function(obs) {
                    return obs.photos && obs.photos.length > 0;
                }).map(function(obs) {
                    return {
                        image_url:       obs.photos[0].url.replace('square', 'medium'),
                        scientific_name: obs.taxon
                                             ? obs.taxon.name
                                             : (obs.species_guess || 'Unknown'),
                        common_name:     obs.taxon && obs.taxon.preferred_common_name
                                             ? obs.taxon.preferred_common_name
                                             : (obs.species_guess || ''),
                        user_login:      obs.user ? obs.user.login : 'unknown'
                    };
                });
        } catch (e) {
            console.warn('iNaturalist fetch failed for:', query, e);
            return [];
        }
    }

    // =====================================================================
    // Global Names API — replaces the R urltospecies() function
    //
    // Original R:
    //   mydata <- jsonlite::stream_in(url(paste0(
    //       "https://gnrd.globalnames.org/name_finder.json?url=", url)))
    //   myspecies <- mydata$names[[1]]$scientificName
    //   countSpaces <- function(s) {
    //       sapply(gregexpr(" ", s), function(p) { sum(p>=0) })
    //   }
    //   myspecies <- myspecies[which(countSpaces(myspecies)==1)]
    //   myspecies <- unique(myspecies)
    // =====================================================================
    async function urlToSpecies(inputUrl) {
        try {
            const resp = await fetch(
                'https://gnrd.globalnames.org/name_finder.json?url=' +
                encodeURIComponent(inputUrl)
            );
            const data = await resp.json();
            if (!data.names || data.names.length === 0) return [];

            // Filter to binomials (exactly one space = genus + species)
            const seen = new Set();
            const results = [];
            for (let i = 0; i < data.names.length; i++) {
                const name = data.names[i].scientificName;
                if (!name) continue;
                const spaceCount = (name.match(/ /g) || []).length;
                if (spaceCount === 1 && !seen.has(name)) {
                    seen.add(name);
                    results.push(name);
                }
            }
            return results;
        } catch (e) {
            console.error('Global Names lookup failed:', e);
            return [];
        }
    }

    // =====================================================================
    // Show a random observation — replaces the Shiny renderText() calls
    //
    // Original R:
    //   i = sample(length(image_list),1)
    //   local_obs <- image_list[[i]]
    //   to_show <- local_obs[sample(sequence(nrow(local_obs)),1),]
    //   output$picture <- renderText({c('<img src="', to_show$image_url,
    //       '" title="', paste0(to_show$common_name, ' (',
    //       to_show$scientific_name, ')'), '">')})
    // =====================================================================
    function showRandomImage() {
        if (imageList.length === 0) {
            pictureEl.innerHTML = '<p>No images loaded yet.</p>';
            creditEl.innerHTML  = '';
            return;
        }
        const speciesIdx = Math.floor(Math.random() * imageList.length);
        const obsArray   = imageList[speciesIdx];
        const obs        = obsArray[Math.floor(Math.random() * obsArray.length)];

        const titleText = obs.common_name + ' (' + obs.scientific_name + ')';
        pictureEl.innerHTML =
            '<img src="' + obs.image_url +
            '" title="' + titleText +
            '" alt="Identify this species">';
        creditEl.innerHTML =
            'Photo by user <a href="https://www.inaturalist.org/people/' +
            obs.user_login + '">' + obs.user_login +
            '</a> submitted to <a href="https://www.inaturalist.org/">iNaturalist.org</a>';
    }

    // =====================================================================
    // "Store image locations" — replaces observeEvent(input$queryButton, …)
    //
    // Original R:
    //   observeEvent(input$queryButton, {
    //       species <- trimws(unlist(strsplit(input$speciesArea, ",")))
    //...
    //       for (i in seq_along(species)) {
    //           obs <- rinat::get_inat_obs(query=species[i], maxresults=50)
    //...
    //       }
    //   })
    // =====================================================================
    queryButton.addEventListener('click', async function() {
        const rawInput = speciesArea.value.trim();
        let speciesList = rawInput.split(',').map(function(s) { return s.trim(); }).filter(function(s) { return s.length > 0; });

        // Single URL detection — replaces: if(length(species)==1) { species <- urltospecies(...) }
        if (speciesList.length === 1 && speciesList[0].startsWith('http')) {
            loadedEl.textContent = 'Extracting species names from URL…';
            speciesList = await urlToSpecies(speciesList[0]);
            if (speciesList.length === 0) {
                loadedEl.textContent =
                    'Could not extract any species names from that URL.';
                return;
            }
            loadedEl.textContent =
                'Found ' + speciesList.length +
                ' species from URL. Now fetching images…';
        }

        // Reset
        imageList = [];
        const badSpecies = [];
        queryButton.disabled = true;
        goButton.disabled    = true;
        introEl.style.display = 'none';
        progressContainer.style.display = 'block';
        progressFill.style.width = '0%';

        for (let i = 0; i < speciesList.length; i++) {
            const species = speciesList[i];
            progressText.textContent = 'Looking up: ' + species;
            loadedEl.textContent     = 'Now looking up ' + species;

            const obs = await getINatObs(species, 50);
            if (obs.length > 0) {
                imageList.push(obs);
            } else {
                badSpecies.push(species);
            }

            const pct = ((i + 1) / speciesList.length) * 100;
            progressFill.style.width = pct + '%';

            // Throttle requests (iNaturalist rate limiting)
            if (i < speciesList.length - 1) {
                await new Promise(function(r) { setTimeout(r, 300); });
            }
        }

        progressContainer.style.display = 'none';
        queryButton.disabled = false;

        let msg = 'Loaded addresses of info for ' + imageList.length + ' species';
        if (badSpecies.length > 0) {
            msg += ' but failed for ' + badSpecies.join(', ');
        }
        loadedEl.textContent = msg;

        if (imageList.length > 0) {
            goButton.disabled = false;
            showRandomImage();
        }
    });

    // =====================================================================
    // "New image!" — replaces observeEvent(input$goButton, …)
    // =====================================================================
    goButton.addEventListener('click', function() {
        loadedEl.textContent = '';
        showRandomImage();
    });

    return 'Flash Cards app initialized';
}
