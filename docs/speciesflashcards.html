Species Flash Cards

Enter species below (common or scientific names, comma-separated), then click New image. Mouseover the image to reveal the answer.


This is made possible due to iNaturalist.org and all their users who make observations and help identify species, but it is in no way affiliated with them; if this app is having issues, please contact Brian O'Meara, not the good iNaturalist folks. Source code: GitHub. 
app = {
    const btn       = document.getElementById('newImageBtn');
    const textarea  = document.getElementById('speciesArea');
    const pictureEl = document.getElementById('picture');
    const creditEl  = document.getElementById('credit');
    const statusEl  = document.getElementById('status');

    btn.addEventListener('click', async function () {
        // Parse species list from textarea
        const speciesList = textarea.value.split(',').map(s => s.trim()).filter(s => s.length > 0);

        if (speciesList.length === 0) {
            statusEl.textContent = 'Please enter at least one species.';
            return;
        }

        // Pick a random species
        const species = speciesList[Math.floor(Math.random() * speciesList.length)];

        statusEl.textContent = 'Loading…';
        pictureEl.innerHTML  = '';
        creditEl.innerHTML   = '';
        btn.disabled = true;

        try {
            const resp = await fetch(
                'https://api.inaturalist.org/v1/observations?' +
                'q=' + encodeURIComponent(species) +
                '&quality_grade=research' +
                '&photos=true' +
                '&per_page=50' +
                '&order=desc&order_by=id'
            );
            const data = await resp.json();

            const withPhotos = (data.results || []).filter(
                obs => obs.photos && obs.photos.length > 0
            );

            if (withPhotos.length === 0) {
                statusEl.textContent =
                    'No research-grade photos found for "' + species + '". Try again!';
                btn.disabled = false;
                return;
            }

            const obs = withPhotos[Math.floor(Math.random() * withPhotos.length)];

            const sciName    = obs.taxon ? obs.taxon.name : (obs.species_guess || 'Unknown');
            const commonName = obs.taxon && obs.taxon.preferred_common_name
                                   ? obs.taxon.preferred_common_name
                                   : '';
            const imageUrl   = obs.photos[0].url.replace('square', 'medium');
            const userLogin  = obs.user ? obs.user.login : 'unknown';

            const titleText = commonName
                ? commonName + ' (' + sciName + ')'
                : sciName;

            pictureEl.innerHTML =
                '<img src="' + imageUrl +
                '" title="' + titleText +
                '" alt="Identify this species">';

            creditEl.innerHTML =
                'Photo by <a href="https://www.inaturalist.org/people/' +
                userLogin + '">' + userLogin +
                '</a> on <a href="https://www.inaturalist.org/">iNaturalist</a>';

            statusEl.textContent = '';
        } catch (e) {
            statusEl.textContent = 'Error fetching data: ' + e.message;
            console.error(e);
        }

        btn.disabled = false;
    });
}
